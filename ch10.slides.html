# plt.bar()
import numpy as np
import matplotlib.pyplot as plt

N = 5
ice_cream_sales = np.random.randint(50, 100, size = 5)
ind = np.arange(N)
width = 0.5
fig, ax = plt.subplots()
ax.bar(ind, ice_cream_sales, width, color = 'g')
ax.set_ylabel('Sales')
ax.set_title('Ice cream sales by flavor')
ax.set_xticks(ind)
ax.set_xticklabels(('Matcha', 'Chocolate', 'Vanilla', 'Coco', 'Strawberry'))
plt.show()

# plt.boxplot()

import numpy as np
import matplotlib.pyplot as plt

data_size = (100, 1)
data_norm = np.random.normal(size = data_size)
data_lognormal = np.random.lognormal(size = data_size)
data_unif = np.random.uniform(size = data_size)
data = np.hstack((data_norm, data_unif, data_unif))
labels = ['Normal', 'Lognormal', 'Uniform']
fig, ax = plt.subplots()
ax.boxplot(data, labels = labels)
ax.set_title('Distribution types')
plt.show()

# 2 X 1
import matplotlib.pyplot as plt
import numpy as np

x = np.linspace(0, 2 * np.pi, 50)
y = np.sin(x)
plt.subplots(2, 1, sharex = True)
plt.subplot(2, 1, 1).plot(x, y)
plt.subplot(2, 1, 1).set_title('Sharing X Axis')
plt.subplot(2, 1, 2).scatter(x, y, color = 'r')
plt.tight_layout()
plt.show()

import numpy as np
